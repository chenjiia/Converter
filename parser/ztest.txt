//8 bit ver
var S = 0;
var O = 0;

var SMax = 256;
var OMax = 256;

var STemp = 0;
var OTemp = 0;
var OCounter = 0;

while(O != OMax){
	S = 0;
	while(S != SMax){
		//code start
		OTemp = 0;
		STemp = S;
		
		STemp= (STemp&85) + ((STemp>>1)&85);
		STemp= (STemp&51) + ((STemp>>2)&51);
		STemp= (STemp + (STemp>>4)&15);

		//code end
		if(O == OTemp){
			OCounter = OCounter + 1;
			goto break;
		}
		S = S + 1;
	}
	break : O = O + 1;
}

if(OCounter == 9){
	reach: 0>1;
}
